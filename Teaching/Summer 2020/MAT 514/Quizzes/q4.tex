\begin{exercise}{1}
Use the relation
\begin{equation*}
\sin(z) = \frac{e^{iz}-e^{-iz}}{2i}
\end{equation*}
to show that $\sin(z)$ is an antiderivative of $\cos(z)$. 
\end{exercise}

\vfill
\begin{exercise}{2}
Compute the derivative of 
\begin{equation*}
\cosh(z^2) 
\end{equation*}
by using properties of derivatives and complex exponentials.
\vfill
\end{exercise}
