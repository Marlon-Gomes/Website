\begin{exercise}{1}
Compute the integral
\begin{equation*}
\int_{\gamma} \frac{z+1}{z}	\, dz,
\end{equation*}
where $\gamma$ is the line segment joining $4$ to $4i$.
\end{exercise}

\end{exercise}
