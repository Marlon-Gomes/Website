\question[4] Consider the initial-value problem
\begin{align*}
y'(x) & = x+y, \\
y(0) & = 1.
\end{align*}
Estimate the value of $y(1)$ using Euler's method with $4$ steps.

\begin{parts}
\part[3] Find $P(t)$.
